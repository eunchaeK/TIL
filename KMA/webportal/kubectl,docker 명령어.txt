-------------------------------------------------
 namespace 해당 pod 조회
-------------------------------------------------
kubectl get pod -n {namespace}


-------------------------------------------------
 pod 로그 조회
-------------------------------------------------
kubectl logs -f -n {namespace} {podname} --since=1m


-------------------------------------------------
 pod 이벤트(A)
-------------------------------------------------
kubectl get event -A


-------------------------------------------------
 docker image 조회 (size 내림차순)
-------------------------------------------------
※CICD 서버에서 조회
(echo -e "REPOSITORY  TAG  SIZE  CREATED" && docker images --format "{{.Repository}}  {{.Tag}}  {{.Size}}  {{.CreatedAt}}" | sort -rh -k 3) | column -t
