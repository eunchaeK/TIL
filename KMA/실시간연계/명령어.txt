---------------------------------------------------------
 TCP 연결 확인 
---------------------------------------------------------
1) echo > /dev/tcp/{IP}/{Port}

-결과 확인
echo $?
0: 성공, 1: 실패

2) tracepath -p {Port} {IP}


---------------------------------------------------------
 TCP port 확인
---------------------------------------------------------
netstat -an | grep {IP or Port}


---------------------------------------------------------
 심볼릭링크 깨진 파일 조회
---------------------------------------------------------
find . -type | ! -exec test -e {}\; -print


---------------------------------------------------------
 심볼릭링크 깨진 파일만 삭제 
---------------------------------------------------------
find . -type | ! -exec test -e {}\; -exec rm -f {} \;
find . -type | ! -exec test -e {}\; -print | xargs rm -f


---------------------------------------------------------
 파일 내용 변경 내용 비교
---------------------------------------------------------
diff -urNp {file1} {file2} | vimdiff - -R



---------------------------------------------------------
 파일 내용 변경 내용 비교
---------------------------------------------------------
※ v 옵션: 압축(해제) 내용 화면 출력

-tar.gz로 디렉토리 압축 
tar -zcf {directory}.tar.gz {directory}

-gz 압축파일 찾기 및 압축해제
find ./ -name "*.gz" | xargs gunzip

-tar.gz 압축해제
tar -zxf {file}.tar.gz
